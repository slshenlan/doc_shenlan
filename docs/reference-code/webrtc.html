import { ref } from 'vue';
import { slGenUid } from 'shenlan-util';
import slPubWebSocket from '@/libs/js/socket';
import config from '@/config/index.js';

// 视频流推送
function webRTCPublish() {
  this.peer = new RTCPeerConnection(null);
  this.stream = new MediaStream();

  this.publish = async (params) => {
    const { url, streamurl } = params;

    this.peer.addTransceiver('audio', { direction: 'sendonly' });
    this.peer.addTransceiver('video', { direction: 'sendonly' });

    const stream = await navigator.mediaDevices.getUserMedia({
      audio: true,
      video: {
        width: { ideal: 320, max: 576 }
      }
    });

    stream.getTracks().forEach((track) => {
      this.peer.addTrack(track);
      this.stream.addTrack(track);
    });

    const offer = await this.peer.createOffer();
    await this.peer.setLocalDescription(offer);

    const answer = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        streamurl,
        sdp: offer.sdp,
        api: url,
        tid: slGenUid(8),
        clientip: null
      })
    }).then((response) => {
      if (!response.ok) {
        throw new Error(`HTTP error! Status: ${response.status}`);
      }
      return response.json();
    });

    await this.peer.setRemoteDescription(
      new RTCSessionDescription({ type: 'answer', sdp: answer.sdp })
    );
    console.log('---推送视频成功---');
  };

  this.close = function () {
    console.log('---关闭webRTCPublish---');

    this.peer && this.peer.close();
    this.peer = null;
  };
}

// 视频流播放
function webRTCPlayer() {
  this.peer = new RTCPeerConnection(null);
  this.stream = new MediaStream();

  this.peer.ontrack = (event) => {
    this.stream.addTrack(event.track);
  };

  this.play = async (params) => {
    this.peer.addTransceiver('audio', { direction: 'recvonly' });
    this.peer.addTransceiver('video', { direction: 'recvonly' });

    const offer = await this.peer.createOffer();
    await this.peer.setLocalDescription(offer);

    const { url, streamurl } = params;

    const answer = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        streamurl,
        sdp: offer.sdp,
        api: url,
        tid: slGenUid(8),
        clientip: null
      })
    }).then((response) => {
      if (!response.ok) {
        throw new Error(`HTTP error! Status: ${response.status}`);
      }
      return response.json();
    });

    await this.peer.setRemoteDescription(
      new RTCSessionDescription({ type: 'answer', sdp: answer.sdp })
    );
  };

  this.close = () => {
    this.peer && this.peer.close();
    this.peer = null;
  };
}

export default function useWebRTCVideo(pageParams) {
  const { host, room, display } = config.socket.video;

  const handleStartPlay = pageParams?.handleStartPlay ? pageParams.handleStartPlay : () => {};
  const handleJoinAndLeave = pageParams?.handleJoinAndLeave
    ? pageParams.handleJoinAndLeave
    : () => {};

  const socket = ref({
    ws: null,
    config: {
      host,
      room,
      display
    }
  });
  async function socketInit() {
    console.log('摄像头websocket地址：', `ws://${host}/sig/v1/rtc?room=${room}&display=${display}`);

    socket.value.ws = new slPubWebSocket({
      url: `ws://${host}/sig/v1/rtc?room=${room}&display=${display}`,

      onMessage: (message) => {
        const msg = message.msg;

        console.log(msg);

        if (msg) {
          if (Object.hasOwnProperty.call(msg, 'event')) {
            const event = msg.event;

            if (event === 'publish') {
              if (msg.peer && msg.peer.publishing && msg.peer.display !== display) {
                // 开始播放
                handleStartPlay(msg.peer.display);
              }
            }

            if (event === 'join' || event === 'leave') {
              // 处理 join 和 leave
              handleJoinAndLeave({
                data: msg.participants,
                devceId: display
              });
            }
          } else {
            if (msg.participants) {
              msg.participants.forEach((ele) => {
                if (ele.display === display || !ele.publishing) return;
                // 开始播放
                handleStartPlay(ele.display);
              });
            }
          }
        }
      },
      // 异常断开重新连接
      onErrorClose: () => {
        console.log('摄像头 socket 断开了');
      }
    });
  }

  // 断开连接
  function webRTCVideoClose() {
    console.log('关闭 socket');
    socket.value.ws && socket.value.ws.close();
  }

  let publisher = null;

  // 开始推送流
  function webRTCStartPublish() {
    if (publisher) {
      publisher.close();
    }
    publisher = new webRTCPublish();
    publisher.publish({
      url: `http://${host}:1985/rtc/v1/publish/`,
      streamurl: `webrtc://${host}/${room}/${display}`
    });
    console.log('推送请求发出了');
  }

  // webRTC 初始化
  async function webRTCVideoInit(deviceCloud) {
    socketInit();

    const ws = socket.value.ws;
    await ws.connect();

    await ws.send({
      action: 'join',
      room: room,
      display: display
    });

    await webRTCStartPublish();

    await ws.send({
      action: 'publish',
      room: room,
      display: display
    });

    if (deviceCloud) {
      deviceCloud();
    }
  }

  // 播放视频流
  function webRTCVideoPlay(params) {
    // streamId  视频流ID
    // videoId 播放器ID
    const { streamId, videoId } = params;
    const player = new webRTCPlayer();

    player.play({
      url: `http://${host}:1985/rtc/v1/play/`,
      streamurl: `webrtc://${host}/${room}/${streamId}`
    });

    const video = document.getElementById(videoId);
    video.srcObject = player.stream;

    return player;
  }

  // 播放摄像头
  function webRTCVideoPlayDeviceCloud(params) {
    // streamId  视频流ID
    // videoId 播放器ID
    const { streamId, videoId } = params;
    const player = new webRTCPlayer();

    console.log('videoId:', videoId);

    player.play({
      url: `http://${host}:1985/rtc/v1/play/`,
      streamurl: `webrtc://${host}/${room}/${streamId}`
      // url: `http://192.168.1.61:1985/rtc/v1/play/`,
      // streamurl: `webrtc://192.168.1.61/live/zr30`
    });

    const video = document.getElementById(videoId);
    video.srcObject = player.stream;

    return player;
  }

  onBeforeUnmount(() => {
    console.log('离开本页面啦-------------------');
    publisher && publisher.close();
    webRTCVideoClose();
  });

  return {
    webRTCVideoInit,
    webRTCVideoPlay,
    webRTCVideoPlayDeviceCloud,
    webRTCVideoClose
  };
}

import useWebRTCVideo from './webRTC-video.js';

export default function useVideo() {
  const { proxy } = getCurrentInstance();
  const { $pub, $api } = proxy;

  const { webRTCVideoInit, webRTCVideoPlay, webRTCVideoPlayDeviceCloud } = useWebRTCVideo({
    handleStartPlay: handleStartPlay,
    handleJoinAndLeave: handleJoinAndLeave
  });

  const videoContentId = $pub.slGenUid();
  const videoDivId = $pub.slGenUid();

  // 视频播放对象
  const videoProps = ref({
    player: null,
    // 视频原大小
    style: {
      width: null,
      height: null
    }
  });

  // 当前是否是全屏
  const isCurrentFullscreen = ref(false);

  // 全屏切换
  function videoFullScreenToggle() {
    isCurrentFullscreen.value = !isCurrentFullscreen.value;
  }

  // 有设备加入 开始播放视频
  function handleStartPlay(id) {
    emits('onJoin', id);
  }

  // 设备离开
  function handleJoinAndLeave(params) {
    console.log('有设备离开了');
    emits('onLeave', params);
  }

  // 播放流视频
  function videoPlay(id) {
    console.log('id:', id);

    const player = webRTCVideoPlay({
      streamId: id,
      videoId
    });

    emits('onPlay', {
      id,
      player
    });
  }
  // 关闭视频
  function videoClose() {
    console.log('videoClose');

    if (videoProps.value.player) {
      console.log('关闭当前视频');

      videoProps.value.player.close();
      videoProps.value.player = null;
    }
  }

  // 一体机摄像头初始化
  function cameraVideoDataInit() {
    webRTCVideoInit(() => {
      // 获取摄像头 ID
      videoProps.value.player = webRTCVideoPlayDeviceCloud({
        streamId: res[0].DeviceNo,
        videoId
      });
    });
  }

  // 计算视频播放窗口大小
  function handleVideoWindow() {
    const videoContent = document.getElementById(videoContentId);
    const videoDiv = document.getElementById(videoDivId);

    const style = videoProps.value.style;
    const videoWidth = style.width;
    const videoHeight = style.height;

    if (videoContent) {
      let videoParentWidth = videoContent.offsetWidth;
      let videoParentHeight = videoContent.offsetHeight;

      if (isCurrentFullscreen.value) {
        // 全屏
        videoParentWidth = window.innerWidth;
        videoParentHeight = window.innerHeight;
      }

      if ((videoWidth * videoParentHeight) / videoParentWidth < videoHeight) {
        // 视频高度超出了
        videoDiv.style.width = `${(videoWidth * videoParentHeight) / videoHeight}px`;
      }
    }
  }

  onMounted(() => {
    nextTick(() => {
      const video = document.getElementById(videoId);

      // 当视频尺寸信息加载完成后
      video.addEventListener('loadedmetadata', function () {
        videoProps.value.style = {
          width: video.videoWidth,
          height: video.videoHeight
        };

        handleVideoWindow();
      });
    });
  });

  onBeforeUnmount(() => {
    console.log('关闭所有视频流，关闭视频socket');

    if (videoProps.value.player) {
      videoProps.value.player.close();
    }
  });

  // 监听 全屏状态 动态计算播放器窗口大小
  watch(
    () => isCurrentFullscreen.value,
    () => {
      nextTick(() => {
        handleVideoWindow();
      });
    }
  );

  return {};
}
