<template>
  <div id="app">
    <keep-alive>
      <router-view v-if="$route.meta.keepAlive && isRouterAlive"> </router-view>
    </keep-alive>

    <transition :name="transitionName">
      <router-view v-if="!$route.meta.keepAlive && isRouterAlive"> </router-view>
    </transition>
  </div>
</template>

<script>
export default {
  name: 'App',
  provide() {
    return {
      reload: this.reload
    };
  },
  data() {
    return {
      isRouterAlive: true,
      transitionName: ''
    };
  },
  watch: {
    $route(to, from) {
      if (to.meta.keepAlive) {
        this.transitionName = '';
      } else {
        if (from.meta.index) {
          if (from.meta.index === to.meta.index) {
            this.transitionName = '';
          } else {
            if (to.meta.index > from.meta.index) {
              this.transitionName = 'slide-left';
              if (from.meta.keepAlive) {
                this.$nextTick(() => {
                  this.transitionName = 'slide-right';
                });
              }
            } else {
              this.transitionName = 'slide-right';
              this.$nextTick(() => {
                this.transitionName = '';
              });
            }
          }
        } else {
          this.transitionName = '';
        }
      }
    }
  },
  methods: {
    reload() {
      this.transitionName = '';
      this.isRouterAlive = false;
      this.$nextTick(function () {
        this.isRouterAlive = true;
      });
    }
  }
};
</script>

// 对象

// 命名的路由
this.$router.push({ name: 'about', params: { id: '123' } }); // /#/about/123

// 带查询的参数
this.$router.push({ path: 'about', query: { id: '123' } }); // /#/about?id=123
this.$router.push({ name: 'about', params: { userId } }); // /#/about/123
this.$router.push({ path: `/about/${userId}` }); // /#//about/123
this.$router.push({ path: '/about', params: { userId } }); // /#/about
